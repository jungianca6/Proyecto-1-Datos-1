158  /**
159   * Permite almacenar información a estos atributos, y
160   * obtener esa información.
161   * El Serializable indica que todas las instancias que
162     pertecenen a esta clase pueden convertirse en una serie de bytes,
163   * los cuales serán enviador a través del Socket
164   */
165  class paqueteDatos implements Serializable{
166      private String nick, ip, mensaje;
167      public String getNick() {
168          return nick;
169      }
170      public void setNick(String nick) {
171          this.nick = nick;
172      }
173      public String getIp() {
174          return ip;
175      }
176      public void setIp(String ip) {
177          this.ip = ip;
178      }
179      public String getMensaje() {
180          return mensaje;
181      }
182      public void setMensaje(String mensaje) {
183          this.mensaje = mensaje;
184      }
185  
186  
187      // Métodos para convertir a JSON y desde JSON
188      public String toJson() {
189          try {
190              ObjectMapper objectMapper = new ObjectMapper();
191              System.out.println(objectMapper.writeValueAsString(this));
192              return objectMapper.writeValueAsString(this);
193          } catch (Exception e) {
194              e.printStackTrace();
195              return null;
196          }
197      }
198  
199      public static paqueteDatos fromJson(String json) {
200          try {
201              System.out.println(json);
202              ObjectMapper objectMapper = new ObjectMapper();
203              return objectMapper.readValue(json, paqueteDatos.class);
204          } catch (Exception e) {
205              e.printStackTrace();
206              return null;
207          }
208      }
